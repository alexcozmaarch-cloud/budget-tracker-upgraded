Connect Budget Tracker
Link to a Google Sheet via Apps Script. A new tab is created each period; a MONTHLY summary sheet tracks your history.
1
Create a Google Sheet

Go to sheets.google.com → new spreadsheet. Name it "Budget Tracker".

2
Open Apps Script

Extensions → Apps Script. Replace everything with the code below, then save.

// Budget Tracker — Apps Script // Paste this entire file, save, then deploy as Web App (Anyone access) function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const action = e.parameter.action; const sheet = e.parameter.sheet || currentSheet_(); if (action === 'getData') return jsonResp(getData_(ss, sheet)); if (action === 'getMonthly') return jsonResp(getMonthly_(ss)); return jsonResp({error:'Unknown action'}); } function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const body = JSON.parse(e.postData.contents); const {action, sheet, type, date, desc, cat, amount, row, budget} = body; if (action === 'addExpense') return jsonResp(addExpense_(ss, sheet, type, date, desc, cat, amount, budget)); if (action === 'deleteExpense') return jsonResp(deleteExpense_(ss, sheet, type, row)); if (action === 'editExpense') return jsonResp(editExpense_(ss, sheet, type, row, date, desc, cat, amount, budget)); return jsonResp({error:'Unknown action'}); } // ─── helpers ────────────────────────────────────────── const CAT_IDS = ['groceries','lunches','transport','running','hobbies','health','media','travel','extras','refund']; function currentSheet_() { const n=new Date(), d=n.getDate(), m=n.getMonth(), y=n.getFullYear(); const mon=['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec']; const startDay = 10; if(d < startDay) { const pm = m===0?11:m-1; const py = m===0?y-1:y; return String(py).slice(-2)+'.'+mon[pm]; } return String(y).slice(-2)+'.'+mon[m]; } function getOrCreateSheet_(ss, name) { let sh = ss.getSheetByName(name); if (!sh) { sh = ss.insertSheet(name); sh.appendRow(['date','desc','cat','amount','running']); sh.setFrozenRows(1); } return sh; } function sheetData_(sh) { if (sh.getLastRow() < 2) return []; const rows = sh.getRange(2,1,sh.getLastRow()-1,5).getValues(); return rows.map((r,i)=>({row:i+2, date:r[0]?Utilities.formatDate(new Date(r[0]),'GMT','yyyy-MM-dd'):'', desc:r[1], cat:r[2], amount:r[3], running:r[4]})); } function calcRunning_(sh) { if (sh.getLastRow() < 2) return; const rows = sh.getRange(2,1,sh.getLastRow()-1,5).getValues(); let run=0; rows.forEach((r,i)=>{ run+=parseFloat(r[3])||0; sh.getRange(i+2,5).setValue(run); }); } function getData_(ss, sheetName) { const ccSh = getOrCreateSheet_(ss, sheetName+'_CC'); const ddSh = getOrCreateSheet_(ss, sheetName+'_DD'); const cc = sheetData_(ccSh); const dd = sheetData_(ddSh); const ccT = cc.reduce((s,r)=>s+(parseFloat(r.amount)||0),0); const ddT = dd.reduce((s,r)=>s+(parseFloat(r.amount)||0),0); return { cc, dd, stats:{ ccTotal:ccT, ddTotal:ddT, total:ccT+ddT }, spreadsheetUrl:ss.getUrl() }; } function addExpense_(ss, sheetName, type, date, desc, cat, amount, budget) { const name = sheetName+'_'+(type==='CC'?'CC':'DD'); const sh = getOrCreateSheet_(ss, name); sh.appendRow([new Date(date), desc, cat||'', parseFloat(amount), 0]); calcRunning_(sh); updateMonthly_(ss, sheetName, budget); return getData_(ss, sheetName); } function deleteExpense_(ss, sheetName, type, row) { const name = sheetName+'_'+(type==='CC'?'CC':'DD'); const sh = ss.getSheetByName(name); if (sh && row > 1) { sh.deleteRow(row); calcRunning_(sh); } updateMonthly_(ss, sheetName); return getData_(ss, sheetName); } function editExpense_(ss, sheetName, type, row, date, desc, cat, amount, budget) { const name = sheetName+'_'+(type==='CC'?'CC':'DD'); const sh = ss.getSheetByName(name); if (!sh || row < 2) return {error:'Row not found'}; sh.getRange(row, 1, 1, 4).setValues([[new Date(date), desc, cat||'', parseFloat(amount)]]); calcRunning_(sh); updateMonthly_(ss, sheetName, budget); return getData_(ss, sheetName); } function getMonthly_(ss) { const sh = ss.getSheetByName('MONTHLY'); if (!sh || sh.getLastRow() < 2) return {data:[]}; const headers = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0]; const rows = sh.getRange(2,1,sh.getLastRow()-1,headers.length).getValues(); return {data: rows.map(r => { const obj = {}; headers.forEach((h,i) => obj[h] = r[i]); return obj; })}; } function updateMonthly_(ss, sheetName, budget) { // Base headers + one col per category const baseHeaders = ['period','budget','cc','dd','total','saving']; const allHeaders = baseHeaders.concat(CAT_IDS); let sh = ss.getSheetByName('MONTHLY'); if (!sh) { sh = ss.insertSheet('MONTHLY'); sh.appendRow(allHeaders); sh.setFrozenRows(1); } // Ensure all category columns exist (migration-safe) const existingHeaders = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0]; CAT_IDS.forEach(id => { if (!existingHeaders.includes(id)) { sh.getRange(1, existingHeaders.length+1).setValue(id); existingHeaders.push(id); } }); const headerRow = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0]; // Compute totals from CC and DD sheets const ccSh = ss.getSheetByName(sheetName+'_CC'); const ddSh = ss.getSheetByName(sheetName+'_DD'); const ccRows = ccSh&&ccSh.getLastRow()>1 ? ccSh.getRange(2,1,ccSh.getLastRow()-1,5).getValues() : []; const ddRows = ddSh&&ddSh.getLastRow()>1 ? ddSh.getRange(2,1,ddSh.getLastRow()-1,5).getValues() : []; const allRows = [...ccRows, ...ddRows]; const ccT = ccRows.reduce((s,r)=>s+(parseFloat(r[3])||0),0); const ddT = ddRows.reduce((s,r)=>s+(parseFloat(r[3])||0),0); const total= ccT+ddT; budget = parseFloat(budget)||1800; // Per-category totals const catTotals = {}; CAT_IDS.forEach(id => catTotals[id] = 0); allRows.forEach(r => { const cat = r[2]; if (cat && catTotals.hasOwnProperty(cat)) { catTotals[cat] += parseFloat(r[3])||0; } }); // Build row in header order const rowData = headerRow.map(h => { if (h==='period') return sheetName; if (h==='budget') return budget; if (h==='cc') return ccT; if (h==='dd') return ddT; if (h==='total') return total; if (h==='saving') return budget-total; return catTotals[h]||0; }); // Upsert const dataRows = sh.getLastRow()>1 ? sh.getRange(2,1,sh.getLastRow()-1,1).getValues() : []; const idx = dataRows.findIndex(r=>r[0]===sheetName); if (idx>=0) sh.getRange(idx+2,1,1,rowData.length).setValues([rowData]); else sh.appendRow(rowData); } function jsonResp(obj){ return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); } // ─── Period-end email summary ───────────────────────────────── // To activate: in Apps Script click clock icon (Triggers) → Add Trigger // Function: sendPeriodSummary | Event source: Time-driven // Type: Month timer | Day of month: 10 | Hour: 08–09 // The trigger runs on the 10th, which is the last day of the closing period. // Change SUMMARY_EMAIL below to your address before deploying. const SUMMARY_EMAIL = 'your@email.com'; // ← change this function sendPeriodSummary() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sh = ss.getSheetByName('MONTHLY'); if (!sh || sh.getLastRow() < 2) return; // Read MONTHLY sheet into objects const headers = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0]; const rows = sh.getRange(2,1,sh.getLastRow()-1,headers.length).getValues(); const data = rows.map(r => { const o={}; headers.forEach((h,i)=>o[h]=r[i]); return o; }); // The period that just ended = most recent row by period string const sorted = [...data].sort((a,b)=>String(b.period).localeCompare(String(a.period))); const cur = sorted[0]; if (!cur) return; const chf = v => 'CHF ' + Math.abs(parseFloat(v)||0).toFixed(2).replace(/\B(?=(\d{3})+(?!\d))/g,"'"); const sign = v => (parseFloat(v)||0) >= 0 ? '+' : '-'; // Category names matching CAT_IDS order const catNames = { groceries:'Groceries', lunches:'Lunches', transport:'Transport', running:'Running', hobbies:'Hobbies', health:'Health', media:'Media & Subs', travel:'Travel', extras:'Extras', refund:'Refund/Income' }; // Build plain-text body const pad = (s, n) => String(s).padEnd(n); let body = '\u25c8 Budget Tracker — Period Summary: ' + cur.period + '\n'; body += '='.repeat(42) + '\n\n'; body += pad('Budget:', 16) + chf(cur.budget) + '\n'; body += pad('Total Spent:', 16) + chf(cur.total) + '\n'; body += pad('vs Budget:', 16) + sign(parseFloat(cur.budget)-parseFloat(cur.total)) + chf(Math.abs((parseFloat(cur.budget)||0)-(parseFloat(cur.total)||0))) + '\n'; body += pad('Net Saving:', 16) + sign(cur.saving) + chf(cur.saving) + '\n'; body += pad('Credit Card:', 16) + chf(cur.cc) + '\n'; body += pad('Debit:', 16) + chf(cur.dd) + '\n'; body += '\n' + '-'.repeat(42) + '\n'; body += 'Category Breakdown\n'; body += '-'.repeat(42) + '\n'; CAT_IDS.forEach(id => { const v = parseFloat(cur[id])||0; if (v > 0) body += pad((catNames[id]||id)+':',18) + chf(v) + '\n'; }); // Running averages (all periods) if (data.length > 1) { const n = data.length; const avgTotal = data.reduce((s,r)=>s+(parseFloat(r.total)||0),0)/n; const avgSaving = data.reduce((s,r)=>s+(parseFloat(r.saving)||0),0)/n; body += '\n' + '-'.repeat(42) + '\n'; body += 'Running Average (' + n + ' periods)\n'; body += '-'.repeat(42) + '\n'; body += pad('Avg Spent:', 16) + chf(avgTotal) + '\n'; body += pad('Avg Saving:', 16) + sign(avgSaving) + chf(avgSaving) + '\n'; } body += '\n' + ss.getUrl() + '\n'; MailApp.sendEmail({ to: SUMMARY_EMAIL, subject: '\u25c8 Budget ' + cur.period + ' — ' + chf(cur.total) + ' spent, ' + chf(cur.saving) + ' saved', body: body }); }

About

Period runs from the 10th of each month to the next 10th.

A new pair of sheet tabs (_CC, _DD) is created automatically each period.

Category budgets are saved locally in your browser.

The Projection bar on the dashboard estimates your end-of-period total based on current daily spend rate.

Recurring Entries

Save common monthly entries locally, then apply them to the current period when needed.

Duplicates for today are skipped automatically
